Base de datos

CREATE TABLE usuario (
    id SERIAL PRIMARY KEY,
    username VARCHAR(50),
    email VARCHAR(100) UNIQUE,
    password TEXT,
    estado BOOLEAN DEFAULT TRUE
);

CREATE TABLE consulta (
    id SERIAL PRIMARY KEY,
    usuario_id INT REFERENCES usuario(id),
    tipo VARCHAR(30),
    parametros TEXT,
    resultado TEXT,
    fecha TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
